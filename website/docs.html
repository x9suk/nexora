Getting Started

Welcome to Nexora! This guide will help you get up and running quickly.

Installation

Install Nexora using your platform's package manager:

# Windows (PowerShell)
irm https://raw.githubusercontent.com/nexora-lang/nexora/main/install.ps1 | iex

# macOS / Linux
curl -fsSL https://raw.githubusercontent.com/nexora-lang/nexora/main/install.sh | sh

Or install manually from GitHub releases:

# Download from: https://github.com/nexora-lang/nexora/releases

Verify your installation:

nexora --version
# Output: Nexora 1.0.0

Hello World

Create your first Nexora program:

// hello.nex
print("Hello, World!")

Run it with:

nexora run hello.nex

Variables

Nexora uses let for mutable variables and const for constants.

// Variable declarations
let name = "Nexora"
let age = 1
const PI = 3.14159

// Type annotations
let count: Int = 42
let message: String = "Hello"
let active: Bool = true

// Type inference
let inferred = 100  // Type is inferred as Int

Data Types

Type Description Example
Int Integer numbers 42
Float Floating-point numbers 3.14
String Text strings "Hello"
Bool Boolean values true / false
Array Ordered collections [1, 2, 3]
Object Key-value pairs { key: "value" }

Functions

Functions in Nexora are first-class citizens.

// Function declaration
func add(a: Int, b: Int): Int {
  return a + b
}

// Arrow function (single expression)
func multiply = (a: Int, b: Int): Int => a * b

// Calling functions
let sum = add(5, 3)       // 8
let product = multiply(4, 2)  // 8

// Default parameters
func greet(name: String = "World"): String {
  return "Hello, ${name}!"
}

greet()          // "Hello, World!"
greet("Nexora")  // "Hello, Nexora!"

Higher-Order Functions

let numbers = [1, 2, 3, 4, 5]

// Map
let doubled = numbers.map(n => n * 2)
// [2, 4, 6, 8, 10]

// Filter
let evens = numbers.filter(n => n % 2 == 0)
// [2, 4]

// Reduce
let sum = numbers.reduce(0, (acc, n) => acc + n)
// 15

Arrays

Arrays are ordered, mutable collections.

// Array declaration
let fruits = ["apple", "banana", "cherry"]
let numbers: Int[] = [1, 2, 3, 4, 5]

// Access elements
let first = fruits[0]   // "apple"
let last = fruits[-1]   // "cherry"

// Array methods
fruits.push("date")
fruits.remove("banana")
fruits.includes("apple")  // true
fruits.length             // 3

// Spread operator
let moreFruits = [...fruits, "elderberry"]

// Destructuring
let [a, b, ...rest] = [1, 2, 3, 4, 5]
// a = 1, b = 2, rest = [3, 4, 5]

Objects

Objects store key-value pairs.

// Object declaration
let user = {
  name: "Alice",
  age: 30,
  email: "alice@example.com"
}

// Access properties
let name = user.name       // "Alice"
let age = user["age"]      // 30

// Add/modify properties
user.role = "admin"
user.age = 31

// Object methods
let person = {
  name: "Bob",
  greet(): String {
    return "Hi, I'm ${this.name}"
  }
}

person.greet()  // "Hi, I'm Bob"

// Spread
let updated = { ...user, role: "superadmin" }

// Destructuring
let { name: userName, age: userAge } = user

Classes

Nexora supports object-oriented programming with classes.

// Class declaration
class Animal {
  name: String
  sound: String

  constructor(name: String, sound: String) {
    this.name = name
    this.sound = sound
  }

  speak(): String {
    return "${this.name} says ${this.sound}!"
  }
}

// Inheritance
class Dog extends Animal {
  constructor(name: String) {
    super(name, "Woof")
  }

  fetch(item: String): String {
    return "${this.name} fetches the ${item}"
  }
}

// Usage
let dog = new Dog("Rex")
dog.speak()      // "Rex says Woof!"
dog.fetch("ball")  // "Rex fetches the ball"

// Static methods
class MathUtils {
  static square(x: Int): Int {
    return x * x
  }
}

MathUtils.square(5)  // 25

Modules

Organize your code with modules.

// math.nex (module file)
export func add(a: Int, b: Int): Int {
  return a + b
}

export func subtract(a: Int, b: Int): Int {
  return a - b
}

export const PI = 3.14159

// main.nex (importing)
import { add, subtract, PI } from "./math"
import * as math from "./math"

add(5, 3)        // 8
math.PI          // 3.14159

Default Exports

// logger.nex
export default func log(message: String): void {
  print("[LOG] ${message}")
}

// main.nex
import log from "./logger"
log("Application started")

Error Handling

Handle errors gracefully with try/catch.

// Throwing errors
func divide(a: Int, b: Int): Int {
  if (b == 0) {
    throw new Error("Division by zero")
  }
  return a / b
}

// Try/catch
try {
  let result = divide(10, 2)
  print(result)  // 5
} catch (err) {
  print("Error: ${err.message}")
}

// Custom errors
class ValidationError extends Error {
  field: String

  constructor(field: String, message: String) {
    super(message)
    this.field = field
  }
}

func validateAge(age: Int): void {
  if (age < 0) {
    throw new ValidationError("age", "Age must be positive")
  }
}

// Finally block
try {
  riskyOperation()
} catch (err) {
  handleError(err)
} finally {
  cleanup()
}
